Create a virtual environment
python3 -m venv env
Activate the environment
source env/bin/activate

Check if the virtual env is in your current project
which python3

Create a main.py

Sample endpoints
@app.get("/greet/{name}")
async def greet(name: str):
    return {"message": f"Welcome {name}"}


# passing query parameter
@app.get("/greet/")
async def greet(name: str):
    return {"message": f"Welcome {name}"}

# mixing path parameter with query parameter
@app.get("/greet/{name}")
async def greet(name: str, age:int):
    return {"message": f"Welcome {name}, you are {age} to day"}


@app.get("/headers", status_code=200)
async def get_headers(
    accept: str = Header(None),
    content_type: str = Header(None),
    user_agent: str = Header(None),
    host: str = Header(None),
):
    request_headers = {}
    request_headers["Accept"] = accept
    request_headers["Content-Type"] = content_type
    request_headers["User-Agent"] = user_agent
    request_headers["Host"] = host
    return request_headers
